function cnt(theform){
	var y=theform;
	var r = 0;
	a=y.replace('\n',' ');
	a=a.split(' ');

}

function getElementsByClassName(strClass, strTag, objContElm) {
  strTag = strTag || "*";
  objContElm = objContElm || document;    
  var objColl = objContElm.getElementsByTagName(strTag);
  if (!objColl.length &&  strTag == "*" &&  objContElm.all) objColl = objContElm.all;
  var arr = new Array();                              
  var delim = strClass.indexOf('|') != -1  ? '|' : ' ';   
  var arrClass = strClass.split(delim);    
  for (var i = 0, j = objColl.length; i < j; i++) {                         
    var arrObjClass = objColl[i].className.split(' ');   
    if (delim == ' ' && arrClass.length > arrObjClass.length) continue;
    var c = 0;
    comparisonLoop:
    for (var k = 0, l = arrObjClass.length; k < l; k++) {
      for (var m = 0, n = arrClass.length; m < n; m++) {
        if (arrClass[m] == arrObjClass[k]) c++;
        if ((delim == '|' && c == 1) || (delim == ' ' && c == arrClass.length)) {
          arr.push(objColl[i]); 
          break comparisonLoop;
        }
      }
    }
  }
  return arr; 
}

<h3>Empire of Silver (Web Only Review)</h3>
<p class="ratd"> Rated <em>M</em></p>
<p><em>Review by David Stratton</em></p>
<p>David only reviewing.<br/><br/>In 1899, as the Boxer Rebellion is starting to create chaos in China, wealthy banker Kang, Zhang Tielin, decides that his third son, Aaron Kwok, will be the one to follow in his father's footsteps. Unfortunately, Third Son is in love with his father's young wife, Hao Lei - they had been sweethearts years before. This creates problems within the banking dynasty.......<br/>Christina Yao's film was actually shot in the north of China six years ago, but was largely re-edited by executive producer Jeremy Thomas into a slick, though rather familiar, period piece. The film encompasses the establishment of a banking empire via the manipulation of the price of salt and the acquisition of a great deal of silver; there's rivalry between bank managers, attacks by bandits on a lonely mountain road and, in the desert, an attack by savage wolves - all of it against a backdrop of family rivalry and political and economic unrest. Locations are wonderfully well used - the film looks very impressive - but the narrative, told in the form of a family history being passed on from an old man to his grandson, is pretty familiar from previous epics of this nature.</p>
